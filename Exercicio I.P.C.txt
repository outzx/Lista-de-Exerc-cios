-->Questão 10:
#include <stdio.h>
#include <math.h>

int main()
{
    float x, y, x2, y2, dist;
    printf("Insira o valor de X e Y do ponto 1: ");
    scanf("%f e %f", &x, &y);
    printf("Agora insira os valores de X e Y do ponto 2: ");
    scanf("%f e %f", &x2, &y2);
    dist = sqrt(pow(x2 - x, 2) + pow(y2 - y, 2));
    printf("A distancia entre os pontos e: %.2f\n", dist);
    
    return 0;
}
